<script>
let currentFolder = null;

function load(folderId=null) {
  google.script.run.withSuccessHandler(render).getFiles(folderId);
}

function render(data) {
  currentFolder = data.id;
  let div = document.getElementById("files");
  div.innerHTML = "";

  // FOLDERS
  data.folders.forEach(f => {
    div.innerHTML += `
      <div class="card" onclick="openFolder('${f.id}')">
        📁 ${f.name}
      </div>
    `;
  });

  // FILES
  data.files.forEach(file => {
    div.innerHTML += `
      <div class="card" onclick="openPreview('${file.name}','${file.preview}')">
        <div class="thumb">
          <iframe src="${file.preview}"></iframe>
        </div>
        <div>${file.name}</div>

        <div class="btns">
          <button class="btn" onclick="event.stopPropagation(); window.open('${file.preview}')">Open</button>
          <button class="btn" onclick="event.stopPropagation(); renameFile('${file.id}')">Rename</button>
          <button class="btn" onclick="event.stopPropagation(); deleteFile('${file.id}')">Delete</button>
        </div>
      </div>
    `;
  });
}

// FOLDER
function openFolder(id) {
  load(id);
}

// PREVIEW PANEL
function openPreview(name, url) {
  document.getElementById("panel").classList.add("active");
  document.getElementById("pname").innerText = name;
  document.getElementById("preview").src = url;
}

// DELETE
function deleteFile(id) {
  if(confirm("Delete?")){
    google.script.run.withSuccessHandler(()=>load(currentFolder)).deleteFile(id);
  }
}

// RENAME
function renameFile(id) {
  let name = prompt("New name:");
  if(name){
    google.script.run.withSuccessHandler(()=>load(currentFolder)).renameFile(id,name);
  }
}

// INIT
load();
</script>
